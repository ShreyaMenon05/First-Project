import pandas as pd
import numpy as np
import datetime
import joblib
import tkinter as tk
from tkinter import ttk, messagebox
from tkcalendar import DateEntry

# Load the trained model
model = joblib.load("flight_rf.pkl")
expected_features = model.feature_names_in_

# Options
airlines = [
    "Air India", "GoAir", "IndiGo", "Jet Airways", "Jet Airways Business",
    "Multiple carriers", "Multiple carriers Premium economy", "SpiceJet",
    "Trujet", "Vistara", "Vistara Premium economy"
]
sources = ["Chennai", "Delhi", "Kolkata", "Mumbai"]
destinations = ["Cochin", "Delhi", "Hyderabad", "Kolkata", "New Delhi"]
stop_options = {"Non-stop": 0, "1 Stop": 1, "2 Stops": 2, "3 Stops": 3, "4 Stops": 4}

def open_popup():
    popup = tk.Toplevel(root)
    popup.title("Enter Flight Details")
    popup.geometry("400x400")
    popup.configure(bg="#E3F2FD")

    user_inputs = {}

    def create_dropdown(label, values):
        frame = tk.Frame(popup, bg="#E3F2FD")
        frame.pack(pady=5, fill=tk.X)
        tk.Label(frame, text=label, bg="#E3F2FD").pack(side=tk.LEFT, padx=10)
        var = tk.StringVar()
        dropdown = ttk.Combobox(frame, textvariable=var, values=values, state="readonly")
        dropdown.pack(side=tk.RIGHT, padx=10)
        var.set("")
        user_inputs[label] = var

    def create_date_picker(label):
        frame = tk.Frame(popup, bg="#E3F2FD")
        frame.pack(pady=5, fill=tk.X)
        tk.Label(frame, text=label, bg="#E3F2FD").pack(side=tk.LEFT, padx=10)
        var = tk.StringVar()
        date_entry = DateEntry(frame, textvariable=var, width=12, background='darkblue', foreground='white', borderwidth=2)
        date_entry.pack(side=tk.RIGHT, padx=10)
        user_inputs[label] = var

    create_dropdown("Select Airline:", airlines)
    create_dropdown("Select Source:", sources)
    create_dropdown("Select Destination:", destinations)
    create_dropdown("Number of Stops:", list(stop_options.keys()))
    create_date_picker("Departure Date:")
    create_date_picker("Arrival Date:")

    result_label = tk.Label(popup, text="", bg="#E3F2FD", font=("Arial", 12, "bold"))
    result_label.pack(pady=10)

    def predict_fare():
        airline = user_inputs["Select Airline:"].get()
        source = user_inputs["Select Source:"].get()
        destination = user_inputs["Select Destination:"].get()
        stops_text = user_inputs["Number of Stops:"].get()
        departure_date = user_inputs["Departure Date:"].get()
        arrival_date = user_inputs["Arrival Date:"].get()

        if not airline or not source or not destination or not stops_text or not departure_date or not arrival_date:
            result_label.config(text="⚠ Please select all fields!", fg="red")
            return

        stops = stop_options[stops_text]
        dep_date = pd.to_datetime(departure_date, format="%m/%d/%y")
        journey_day = dep_date.day
        journey_month = dep_date.month

        input_data = pd.DataFrame(columns=expected_features)
        input_data.loc[0] = [0] * len(expected_features)
        input_data["Total_Stops"] = stops
        input_data["Journey_day"] = journey_day
        input_data["Journey_month"] = journey_month

        if f"Airline_{airline}" in input_data.columns:
            input_data[f"Airline_{airline}"] = 1
        if f"Source_{source}" in input_data.columns:
            input_data[f"Source_{source}"] = 1
        if f"Destination_{destination}" in input_data.columns:
            input_data[f"Destination_{destination}"] = 1

        predicted_fare = model.predict(input_data)[0]
        result_label.config(text=f"✈ Estimated Flight Fare: ₹{round(predicted_fare, 2)}", fg="blue")

    predict_button = tk.Button(popup, text="Predict Fare", command=predict_fare, bg="#64B5F6", fg="white", font=("Arial", 12, "bold"))
    predict_button.pack(pady=10)

root = tk.Tk()
root.title("Flight Fare Predictor")
root.geometry("300x200")
root.configure(bg="#E3F2FD")

open_popup_button = tk.Button(root, text="Enter Flight Details", command=open_popup, bg="#64B5F6", fg="white", font=("Arial", 12, "bold"))
open_popup_button.pack(pady=50)

root.mainloop()
